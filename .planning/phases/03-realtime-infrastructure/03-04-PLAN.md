2. **Create hook:**
   ```typescript
   export function useRoomPresence(roomId: string) {
     const {
       setRoomId,
       addParticipant,
       removeParticipant,
       setConnected,
       isConnected
     } = useRoomStore()

     useEffect(() => {
       // Don't connect if already connected to this room
       if (isConnected && useRoomStore.getState().roomId === roomId) {
         return
       }

       // Connect to room namespace
       const socket = connectToRoom(roomId)

       // Connection handlers
       socket.on('connect', () => {
         console.log('useRoomPresence: Connected to room:', roomId)
         setConnected(true)
         setRoomId(roomId)

         // Send heartbeat every 15 seconds
         const heartbeatInterval = setInterval(() => {
           if (socket.connected) {
             socket.emit('heartbeat', { roomId, timestamp: Date.now() })
           }
         }, 15000) // Half of server's 30s interval

         // Cleanup function for this interval
         return () => clearInterval(heartbeatInterval)
       })

       socket.on('disconnect', () => {
         console.log('useRoomPresence: Disconnected from room:', roomId)
         setConnected(false)
       })

       // Presence events
       socket.on('user-joined', (data) => {
         addParticipant({
           userId: data.userId,
           userName: data.userName,
           userPhoto: data.userPhoto
         })
       })

       socket.on('user-left', (data) => {
         removeParticipant(data.userId)
       })

       socket.on('presence-update', (data) => {
         // Full participant list update (for reconnection sync)
         setParticipants(data.participants.map((p: any) => ({
           userId: p.userId || p,
           userName: p.userName || '',
           userPhoto: p.userPhoto
         })))
       })

       // Cleanup on unmount
       return () => {
         socket.off('connect')
         socket.off('disconnect')
         socket.off('user-joined')
         socket.off('user-left')
         socket.off('presence-update')
       }
     }, [roomId, isConnected, setRoomId, addParticipant, removeParticipant, setConnected])

     const reconnect = useCallback(() => {
       // Force reconnection by clearing state
       const { disconnectFromRoom } = require('@/lib/socket')
       disconnectFromRoom(roomId)
       setConnected(false)
       useRoomStore.getState().reset()
     }, [roomId, setConnected])

     return {
       isConnected,
       reconnect
     }
   }
   ```
